\section*{1. Calcular la ganancia en dB de un amplificador en el cual se duplica la potencia
de salida con respecto a la potencia de entrada}
La forma de expresar la ganancia en dB para la potencia está dada por la relación
\begin{equation}
    P(dB) = 10\log_{10} \left( \frac{P_{out}}{P_{in}}\right)
\end{equation}
Debido a que el amplificador entrega a la salida el doble de potencia que en la entrada

\[
   \begin{gathered} 
       P_{out} = 2 P_{in} \\
       P(dB) = 10\log_{10} \left( \frac{2P_{in}}{P_{in}}\right) \\
       \boxed{P(dB) = \SI{3{,}010}{\decibel}\approxeq \SI{3}{\decibel}}
   \end{gathered} 
\]

\section*{2. Tenemos un canal con una atenuación de 10 dB. ¿Cuál es la potencia de salida
(dBm y W) de la señal al final del canal, si la potencia de entrada son 5 W? }


\subsection*{Cálculo en dBm}
La unidad dBm hace referencia a la relación entre la potencia y
\SI{1}{\milli\watt}. Para \SI{5}{\watt}, es decir \SI{5000}{\milli\watt}, de potencia de
entrada

\begin{equation}
    P(dBm) = 10\log_{10} \left( \frac{P}{\SI{1}{\milli\watt}}\right)
    \label{eq:dbm}
\end{equation}

\[
   \begin{gathered} 
       P_{in}(dBm) = 10\log_{10} \left(
       \frac{\SI{5000}{\milli\watt}}{\SI{1}{\milli\watt}}\right) \\
       P_{in}(dBm) = \SI{36{,}99}{\dBm} \approxeq \SI{37}{\dBm}
   \end{gathered} 
\]
Debido a que el canal posee una atenuación de \SI{10}{\decibel}
\[
   \begin{gathered} 
       P_{out}(dBm) = \SI{37}{\dBm} - \SI{10}{\decibel} \\
       \boxed{P_{out}(dBm) = \SI{27}{\dBm}}
   \end{gathered} 
\]

\newpage
\subsection*{Cálculo en W}
Si despejamos la variable P de la ecuación~\eqref{eq:dbm}
\[
        \frac{P(dBm)}{10} = \log_{10} \left(\frac{P}{\SI{1}{\milli\watt}}\right)\\[5mm]
\]
\begin{equation}
    P = 10^{\frac{dBm}{10}}\ [mW]
    \label{eq:dbMtoP}
\end{equation}

Sustituyendo dBm en \eqref{eq:dbMtoP}

\[
    \begin{gathered}
        P_{out} = 10^{\frac{27}{10}} \\
        \boxed{P_{out} = \SI{501{,}19}{\milli\watt}}
    \end{gathered}
\]

\section*{3. Calcular la potencia en dBm de un amplificador cuya salida es de 400 mW}
Utilizando la expresión \eqref{eq:dbm}
\[
    \begin{gathered}
        P_{out}(dBm) = 10 \log_{10} \left(\frac{\SI{400}{\milli\watt}}{\SI{1}{\milli\watt}}\right) \\
        \boxed{P_{out}(dBm) = \SI{26{,}02}{\dBm}}
    \end{gathered}
\]

\section*{4. Calcular la potencia de entrada del amplificador del problema anterior (dBm),
sabiendo que su ganancia es de 20 dB}
Debido a que la ganancia del amplificador del punto anterior es \SI{20}{\decibel}, la
potencia de entrada será

\[
    \begin{gathered}
        P_{in}(dBm) = P_{out}(dBm) - \SI{20}{\decibel} \\
        P_{in}(dBm) = \SI{26{,}02}{\dBm} - \SI{20}{\decibel} \\
        \boxed{P_{in}(dBm) = \SI{6{,}02}{\dBm}}
    \end{gathered}
\]

\section*{5. Para el mismo amplificador si la potencia de entrada es de 1 W, calcular la
potencia de salida en dBW y dBm}
\begin{itemize}
    \item \underline{Cálculo en dBW}

        \vspace{2mm}
La unidad dBW es la relación logarítmica entre la potencia del sistema y \SI{1}{\watt},
de esta forma se obtiene
\begin{equation}
    P(dBW) = 10 \log_{10} \left(\frac{P(W)}{\SI{1}{\watt}}\right)
    \label{eq:dbw}
\end{equation}

Por lo tanto, reemplazando una potencia de \SI{1}{\watt} en \eqref{eq:dbw}

\[
    \begin{gathered}
        P_{in}(dBW) = 10 \log{10} \left(\frac{\SI{1}{\watt}}{\SI{1}{\watt}}\right) \\
        P_{in}(dBW) = \SI{0}{\decibel\watt}
    \end{gathered}
\]

Si el amplificador tiene una ganancia de \SI{20}{\decibel}, entonces

\[
    \boxed{P_{out}(dBW) = \SI{20}{\decibel\watt}}
\]

\item \underline{Cálculo en dBm}

    \vspace{2mm}
    La potencia de entrada en dBm es
    \[
        \begin{gathered}
            P_{in}(dBm) = 10 \log{10} \left(\frac{\SI{1000}{\milli\watt}}{\SI{1}{\milli\watt}}\right) \\
            P_{in}(dBm) = \SI{30}{\dBm}
        \end{gathered}
    \]
    
    Por lo tanto a la salida del amplificador
    \[
        \begin{gathered}
            P_{out}(dBm) = \SI{30}{\dBm} + \SI{20}{\decibel} \\
            \boxed{P_{out}(dBm) = \SI{50}{\dBm}}
        \end{gathered}
    \]

\end{itemize}

\section*{6. Pasar los siguientes valores -73 dBm, -76 dBm, -79 dBm, -77 dBm y -71 dBm a
pW}

Para convertir la unidad dBm a W se utilizará la siguiente ecuación

\begin{equation}
    P(W) = 10^{\frac{P(dBm) - 30}{10}}
    \label{eq:dbmToW}
\end{equation}

Reemplazando los valores dados en \eqref{eq:dbmToW}

\begin{itemize}
    \item \underline{P = \SI{-73}{\dBm}}
        \[
            \begin{gathered}
                P(W) = 10^{\frac{-73 - 30}{10}} \\
                \boxed{P(W) = \SI{50{,}11}{\pico\watt}}
            \end{gathered}
        \]
    \item \underline{P = \SI{-76}{\dBm}}
        \[
            \begin{gathered}
                P(W) = 10^{\frac{-76 - 30}{10}} \\
                \boxed{P(W) = \SI{25{,}11}{\pico\watt}}
            \end{gathered}
        \]
    \item \underline{P = \SI{-79}{\dBm}}
        \[
            \begin{gathered}
                P(W) = 10^{\frac{-79 - 30}{10}} \\
                \boxed{P(W) = \SI{12{,}59}{\pico\watt}}
            \end{gathered}
        \]
    \item \underline{P = \SI{-77}{\dBm}}
        \[
            \begin{gathered}
                P(W) = 10^{\frac{-77 - 30}{10}} \\
                \boxed{P(W) = \SI{20}{\pico\watt}}
            \end{gathered}
        \]
        \newpage
    \item \underline{P = \SI{-71}{\dBm}}
        \[
            \begin{gathered}
                P(W) = 10^{\frac{-71 - 30}{10}} \\
                \boxed{P(W) = \SI{79{,}43}{\pico\watt}}
            \end{gathered}
        \]
\end{itemize}

1000 km, sensibilidad del sistema regenerador de -30 dBm y potencia de salida de cada

\subsection*{b. ¿A qué distancia máxima puede estar la primera estación regeneradora de
señal?}
Se transmite la señal con \SI{100}{\milli\watt}, utilizando la ecuación \eqref{eq:dbm}
\[
    \begin{gathered}
        P_{tx}(dBm) = 10\log_{10} \left(
        \frac{\SI{100}{\milli\watt}}{\SI{1}{\milli\watt}}\right) \\
        P_{tx}(dBm) = \SI{20}{\dBm}
    \end{gathered}
\]
Debido a que el sistema tiene una atenuación de \SI{0{,}17}{\decibel\per\kilo\meter} y que
la sensibilidad del sistema de regeneración es \SI{-30}{\dBm}, la distancia máxima a la
que debe estar el primer regenerador de señal se encuentra cuando dicha señal se atenúa
\SI{50}{\decibel}.

\[
    \begin{gathered}
        d = \frac{\SI{50}{\decibel}}{\SI{0{,}17}{\decibel\per\kilo\meter}} \\
        \boxed{d = \SI{294{,}11}{\kilo\meter}}
    \end{gathered}
\]

El primer regenerador de señal se debe encontrar como máximo a una distancia de
\SI{294{,}11}{\kilo\meter} del transmisor.

\subsection*{c. ¿Cuántas estaciones regeneradoras de señal deben ponerse?}
Al tener \SI{1000}{\kilo\meter} de distancia, con la distancia máxima calculada en el
punto anterior es posible hallar la cantidad de regeneradores de señal necesarios que se
deben colocar
\[
    \begin{gathered}
        n = \frac{\SI{1000}{\kilo\meter}}{\SI{294{,}11}{\kilo\meter}} \\
        \boxed{n = 3,4 \approxeq 4}
    \end{gathered}
\]

Es necesario colocar 4 regeneradores en el sistema.

\subsection*{d. ¿a qué distancia deben estar las estaciones regeneradoras de señal para que
estén equidistantes?}
Debido a que se necesitan 4 estaciones regeneradoras de señal
\[
    \begin{gathered}
        d = \frac{\SI{1000}{\kilo\meter}}{4} \\
        \boxed{d = \SI{250}{\kilo\meter}}
    \end{gathered}
\]
Las estaciones regeneradoras deben colocarse cada \SI{250}{\kilo\meter}

\subsection*{e. Para la distancia obtenida en c, ¿qué potencia en μW y en dBm llega al
receptor?}
La última estación regeneradora se encuentra a \SI{250}{\kilo\meter} del receptor, por lo
tanto, teniendo en cuenta que la atenuación del sistema es
\SI{0{,}17}{\decibel\per\kilo\meter} y que la estación entrega \SI{20}{\dBm}
\[
    \begin{gathered}
        \SI{250}{\kilo\meter} \cdot \SI{0{,}17}{\decibel\per\kilo\meter} =
        \SI{42{,}5}{\decibel} \\
        P_{rx}(dBm) = \SI{20}{\dBm} - \SI{42{,}5}{\decibel} \\
        \boxed{P_{rx}(dBm) = \SI{-22{,}5}{\dBm}}
    \end{gathered}
\]
Utilizando la ecuación \eqref{eq:dbmToW}
\[
    \begin{gathered}
        P_{rx}(W) = 10^{\frac{\SI{-22{,}5}{\dBm} - 30}{10}} \\
        \boxed{P_{rx}(W) = \SI{5{,}62}{\micro\watt}}
    \end{gathered}
\]

\section*{8. El cuadripolo ¿es un amplificador o un atenuador de potencia?}
Para responder la pregunta, es necesario identificar la potencia de entrada y la de salida
\[
    \begin{gathered}
        P_{in} = \frac{V^2}{Z} = \frac{\SI{0{,}5}{\volt}}{\left(\SI{75}{\ohm} \right)^2} \\
        P_{in} = \SI{44{,}44}{\micro\watt} \\[5mm]
        P_{out} = \frac{V^2}{Z} = \frac{\SI{1}{\volt}}{\left(\SI{600}{\ohm} \right)^2} \\
        P_{out} = \SI{2{,}77}{\micro\watt}
    \end{gathered}
\]
Debido a que P\textsubscript{in} es mayor a P\textsubscript{out}, el cuadripolo se trata
de un atenuador.

\section*{9. Dado un canal de transmisión de datos coaxial con una atenuación a la
frecuencia de operación de 0.8dB/100 metros y donde la sensibilidad del
receptor es de -10 dBm. Calcular la potencia mínima que deberá tener el
transmisor si la longitud del coaxial es de 1200 metros. Indicar cuál es la
potencia mínima del receptor en mW}
Si la atenuación de datos en el coaxial es \SI{0{,}8}{\decibel} cada \SI{100}{\metre} y
el receptor puede detectar como mínimo \SI{-10}{\dBm}, entonces para una longitud de cable
de \SI{1000}{\metre}
\[
    \begin{gathered}
        Perdidas = \SI{0{,}8}{\decibel} \cdot 12 \\
        Perdidas = \SI{-9{,}6}{\decibel}
    \end{gathered}
\]
Esto indica que
\[
    \begin{gathered}
        P_{min}(dBm) = \SI{-10}{\dBm} + \SI{9{,}6}{\decibel} \\
        P_{min}(dBm) = \SI{-0{,}4}{\dBm}
    \end{gathered}
\]
Aplicando la ecuación \eqref{eq:dbmToW}
\[
    \begin{gathered}
        P_{mintx}(W) = 10^{\frac{\SI{-0{,}4}{\dBm} - 30}{10}} \\
        \boxed{P_{mintx}(W) = \SI{0{,}912}{\milli\watt}}
    \end{gathered}
\]

La potencia mínima del receptor está dada por su sensibilidad
\[
    \begin{gathered}
        P_{minrx}(W) = 10^{\frac{\SI{-10}{\dBm} - 30}{10}} \\
        \boxed{P_{minrx}(W) = \SI{0{,}1}{\milli\watt}}
    \end{gathered}
\]
